def plot_income_sources(data):
    income_sources = ['fin_inc_sal', 'fin_inc_buss', 'fin_inc_rem', 'fin_inc_pen', 'fin_inc_grant', 
                      'fin_inc_agric', 'fin_inc_none', 'fin_inc_oth']
    income_labels = ['Salaries/Wages', 'Business Income', 'Remittances', 'Pensions', 'Grants', 
                     'Agricultural Sales', 'No Income', 'Other Income']
    
    income_counts = data[income_sources].apply(lambda x: (x == 1).sum())
    
    plt.figure(figsize=(12, 6))
    plt.bar(income_labels, income_counts, color='skyblue')
    plt.xlabel('Income Sources')
    plt.ylabel('Number of Households')
    plt.title('Distribution of Income Sources')
    plt.xticks(rotation=45)
    plt.show()

plot_income_sources(income_exp_data)

def plot_income_sources_pie(data):
    income_sources = ['fin_inc_sal', 'fin_inc_buss', 'fin_inc_rem', 'fin_inc_pen', 'fin_inc_grant', 
                      'fin_inc_agric', 'fin_inc_none', 'fin_inc_oth']
    income_labels = ['Salaries/Wages', 'Business Income', 'Remittances', 'Pensions', 'Grants', 
                     'Agricultural Sales', 'No Income', 'Other Income']
    
    income_counts = data[income_sources].apply(lambda x: (x == 1).sum())
    
    plt.figure(figsize=(10, 10))
    plt.pie(income_counts, labels=income_labels, autopct='%1.1f%%', startangle=140, colors=plt.cm.Paired.colors)
    plt.title('Proportion of Households with Different Income Sources')
    plt.axis('equal')
    plt.show()

def plot_income_expenditure(data):

    ax[0].set_ylabel('Number of Households')
    ax[0].set_title('Distribution of Net Household Income')

    ax[1].set_xlabel('Household Expenditure (Categories)')
    ax[1].set_ylabel('Number of Households')
